1.命令行创建configmap
[root@k8s01 configmap]# kubectl create configmap nginx-config --from-literal=nginx_port=8080 --from-literal=server_name=myserver

2. 从文件创建configmap
[root@k8s01 configmap]# kubectl create configmap cm-user-pwd --from-file=mypwd=/etc/passwd
configmap/cm-user-pwd created
